from django.db import models
from django.contrib.auth.models import AbstractUser
from django.utils import timezone
import uuid
# 扩展Django内置用户模型
class User(AbstractUser):
    """ 
    继承Django AbstractUser扩展用户模型，用于登录注册 
    新增：手机号、头像、用户唯一标识、注册时间等扩展字段 
    """
    # 新增核心扩展字段
    phone = models.CharField(
        max_length=11,
        blank=True,
        null=True,
        unique=True,
        verbose_name="手机号"
    )  # 手机号唯一，支持用手机号登录
    avatar = models.ImageField(
        upload_to="avatars/%Y/%m/%d/",
        blank=True,
        null=True,
        verbose_name="用户头像"
    )  # 头像上传路径按日期分目录
    user_uuid = models.UUIDField(
        default=uuid.uuid4,
        editable=False,
        unique=True,
        verbose_name="用户唯一标识"
    )  # 唯一UUID，避免用自增ID暴露用户数量
    register_time = models.DateTimeField(
        default=timezone.now,
        verbose_name="注册时间"
    )  # 注册时间，默认当前时间
    last_login_ip = models.CharField(
        max_length=50,
        blank=True,
        null=True,
        verbose_name="最后登录IP"
    )  # 记录最后登录IP，可选

    # 元数据配置
    class Meta:
        db_table = "user"  # 数据库表名，简洁易识别
        verbose_name = "用户"
        verbose_name_plural = "用户"  # 后台显示复数名称
        ordering = ["-register_time"]  # 默认按注册时间倒序排列

    def __str__(self):
        """后台/admin显示用户名，而非默认的User object"""
        return self.username

# 可选：用户登录日志模型（记录登录行为，便于排查问题）
class UserLoginLog(models.Model):
    """用户登录日志，用于记录登录行为、排查登录问题"""
    user = models.ForeignKey(
        User,
        on_delete=models.CASCADE,
        related_name="login_logs",
        verbose_name="关联用户"
    )
    login_time = models.DateTimeField(auto_now_add=True, verbose_name="登录时间")
    login_ip = models.CharField(max_length=50, verbose_name="登录IP")
    login_device = models.CharField(
        max_length=100,
        blank=True,
        null=True,
        verbose_name="登录设备"
    )  # 比如：Chrome/Windows 10、iPhone/iOS 17
    login_status = models.BooleanField(default=True, verbose_name="登录状态")  # True=成功，False=失败

    class Meta:
        db_table = "user_login_log"
        verbose_name = "用户登录日志"
        verbose_name_plural = "用户登录日志"
        ordering = ["-login_time"]

    def __str__(self):
        return f"{self.user.username} - {self.login_time.strftime('%Y-%m-%d %H:%M:%S')}"
